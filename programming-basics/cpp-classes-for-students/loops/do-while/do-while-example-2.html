2nd DO-WHILE EXAMPLE

РЕШЕНИЕ № 1 на пример № 2 за DO-WHILE


// Да се отпечатат четните числа от 1 до 15
#include <iostream>
using namespace std ;
main ()
{
  int i=1 ;

  do
    if (0==i%2)
    cout <<i <<endl ;
  while (++i<=15) ;

 }

// Да се отпечатат четните числа от 1 до 15
#include <iostream>
using namespace std ;
main ()
{
  int k = 1 ;

  do
    if (1 == k % 2)
      continue ;
    else
      cout <<k <<endl ;
  while (k++ < 15) ;

 }
