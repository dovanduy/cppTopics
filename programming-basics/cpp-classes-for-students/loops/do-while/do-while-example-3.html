3rd DO-WHILE EXAMPLE

РЕШЕНИЕ № 1
пример № 3 за DO-WHILE


// Да се отпечатат n на брой звездички
#include <iostream>
using namespace std ;
main ()
{
  int i=1, n ;
  cin >>n ;
  do
    cout <<"*" ;
  while (i++ < n) ;
 }

пример № 3 за DO-WHILE


// Да се отпечатат n на брой звездички
#include <iostream>
using namespace std ;
main ()
{
  int n ;
  cin >>n ;
  do
    cout <<"*" ;
  while (--n) ;
 }
