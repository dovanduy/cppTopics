3rd WHILE EXAMPLE

РЕШЕНИЕ № 1 на пример № 3 за WHILE


// Да се отпечатат n на брой звездички
#include <iostream>
using namespace std ;
main ()
{
  int i=1, n ;
  cin >>n ;
  while (i++ <= n)
    cout <<"*" ;
  }
 }

РЕШЕНИЕ № 2 на пример № 3 за WHILE


// Да се отпечатат n на брой звездички
#include <iostream>
using namespace std ;
main ()
{
  int n ;
  cin >>n ;
  while (n--)
    cout <<"*" ;
  }
 }
