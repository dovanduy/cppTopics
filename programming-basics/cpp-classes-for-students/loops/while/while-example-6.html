РЕШЕНИЕ на пример 6 за WHILE

// Да се отпечата най-малката цифра на n
#include <iostream>
using namespace std ;
main ()
{ int n, lastDigit, min ;
  cout <<"n=" ; cin >>n ;
  n = (n<0) ? -n : n ;

  min = lastDigit = n % 10 ;
  while (n = n/10)
  { lastDigit = n % 10 ;
    min = (lastDigit<min) ? lastDigit : min ;
  }
  cout <<"\tmin=" <<min <<endl ;
}
